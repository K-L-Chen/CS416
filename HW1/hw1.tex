\textbf{Question 3} 
\begin{enumerate}
	\item[(1)]
		\[
		\left[
			{\begin{array}{cc}
		   		3 & -1 \\
				2 & 5 \\
				-2 & 2
		  \end{array} } 
		\right]
		\cdot
		\left[
			{\begin{array}{cc}
		   		u & a \\
				v & b
		  \end{array} } 
		\right]
		=
		\left[
			{\begin{array}{cc}
		   		3u - v & 3a - b \\
				2u + 5v & 2a+5b \\
				-2u + 2v & -2a + 2b
		  \end{array} } 
		\right]
		\]
	\item[(2)] Yes, product $AB \in \R^{2\times4}$
	\item[(3)] No.
	\item[(4)]	 $y^TA$ is a row vector, $y^TA \in \R^{1\times2}$
	\item[(5)] $Ax$ is a column vector, $Ax \in \R^3$
	\item[(6)] Since we know that the zero vector $0$ is a row vector, we can proceed as follows:
		\begin{align*}
			(Bx + y)^TA^T &= 0 &&\\
			(Bx + y)^TA^T\cdot (A^T)^-1 &= 0\cdot(A^T)^-1 &&\\
			(Bx + y)^T(A^T\cdot (A^T)^-1) &= 0\cdot(A^T)^-1 &&\\
			(Bx + y)^T(I^{n\times n}) &= 0 &&\\
			(Bx + y)^T &= 0 &&\\
			((Bx + y)^T)^T &= 0^T &&\\
			Bx + y &= 0^T &&\\
			Bx &= 0^T - y &&\\
			Bx &= -y &&\\
			B^{-1}\cdot Bx &= B^{-1}\cdot(-y) &&\\
			(B^{-1}\cdot B)x &= -B^{-1}y &&\\
			I^{n\times n}x &= -B^{-1}y &&\\
			x &= -B^{-1}y
		\end{align*}
	and we are done.
\end{enumerate}
